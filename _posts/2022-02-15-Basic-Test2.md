## 나의 풀이
```javascript
/*** 2. 두 수 사이 숫자 ***/

/* user code */
function answer(x, y) {
  let result = [];

  // 코드 구현 시작 영역
  for (let a = x; a <= y; a++) {
    result.push(a);
  }

  for (let a = y; a <= x; a++) {
    result.push(a);
  }
  // …

  // 코드 구현 종료 영역

  return result;
}

/* main code */
let input = [
  // TC: 1
  [3, 7],
  // TC: 2
  [8, 3],
  // TC: 3
  [12, 10],
];
for (let i = 0; i < input.length; i++) {
  process.stdout.write(`#${i + 1} `);
  console.log(answer(input[i][0], input[i][1]));
}

```

## 해답
for문 2개를 쓰지않고 if로 변수를 스왑해서 문제 해결
```javascript
/*** 2. 두 수 사이 숫자 ***/

/* user code */
function answer(x, y) {
  let result = [];

  if (x > y) {
    let t = x;
    x = y;
    y = t; // 변수 스왑법
  }


  for (let i = x; i <= y; i++) {
    result.push(i);
  }

  return result;
}

/* main code */
let input = [
  // TC: 1
  [3, 7],
  // TC: 2
  [8, 3],
  // TC: 3
  [12, 10],
];
for (let i = 0; i < input.length; i++) {
  process.stdout.write(`#${i + 1} `);
  console.log(answer(input[i][0], input[i][1]));
}
```
